错误：
```
class Solution {
public:
    int candy(vector<int>& ratings) {
        int sum=ratings.size();
        for(int i=1;i<ratings.size();i++)
        {
            if(ratings[i]!=ratings[i-1])sum++;
            if(i<ratings.size()-1&&((ratings[i-1]<ratings[i]&&ratings[i]>ratings[i+1])))sum--;
        }
        return sum;
    }
};
```
